TestCheckIfFileExists tests the functionality of function checkIfFileExists

 
func
 
TestCheckIfFileExists
(
t
 
*
testing
.
T
)
 
{

	
err
 
:=
 
conf
.
CheckIfFileExists
(
""
)

	
if
 
err
 
==
 
nil
 
{

		
t
.
Fatal
(
"File with empty name should not exists"
)

	
}


	
err
 
=
 
conf
.
CheckIfFileExists
(
"config.toml"
)

	
if
 
err
 
!=
 
nil
 
{

		
t
.
Fatal
(
"File should exists:"
,
 
err
)

	
}


	
err
 
=
 
conf
.
CheckIfFileExists
(
"\n"
)

	
if
 
err
 
==
 
nil
 
{

		
t
.
Fatal
(
"File '' should not exist"
)

	
}


	
err
 
=
 
conf
.
CheckIfFileExists
(
"."
)

	
if
 
err
 
==
 
nil
 
{

		
t
.
Fatal
(
"File '.' is a directory"
)

	
}


	
err
 
=
 
conf
.
CheckIfFileExists
(
".."
)

	
if
 
err
 
==
 
nil
 
{

		
t
.
Fatal
(
"File '..' is a directory"
)

	
}


}
