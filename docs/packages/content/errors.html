Package content contains logic for parsing rule content.

 
package
 
content



import
 
"fmt"

MissingMandatoryFile is an error raised while parsing, when a mandatory file is missing

 
type
 
MissingMandatoryFile
 
struct
 
{

	
FileName
 
string


}

InvalidItem is an error raised when unexpected type is found when parsing

 
type
 
InvalidItem
 
struct
 
{

	
FileName
 
string

	
KeyName
  
string


}



func
 
(
err
 
MissingMandatoryFile
)
 
Error
(
)
 
string
 
{

	
return
 
fmt
.
Sprintf
(
"Missing required file: %s"
,
 
err
.
FileName
)


}



func
 
(
err
 
InvalidItem
)
 
Error
(
)
 
string
 
{

	
return
 
fmt
.
Sprintf
(
"Invalid item `%s` of file %s"
,
 
err
.
KeyName
,
 
err
.
FileName
)


}
